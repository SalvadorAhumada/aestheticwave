Adding a class

    
    function addClass(){
    var element = document.getElementById("myDIV");
    element.classList.add("mystyle");
    }

Removing a class

    
    function RemoveClass(){
    var element = document.getElementById("myDIV");
    element.classList.remove("mystyle");
    }

Toggle classes

    
    function toggleClass(){
    var element = document.getElementById("myDIV");
